# Vault Complex
# Unique building vault for tombstart mod 
# Adds +3 science and increases research speed (possibly will change)
building_tombstart_vault = {
	is_listed = no
	icon = "building_primitive_farm"
	
	cost = {
	}

	required_resources = {
	}
	
	produced_resources = {
		physics_research = 3
		society_research = 3
		engineering_research = 3
	}

	planet_modifier = {
	#	pop_environment_tolerance = 0.1
	}

		all_technology_research_speed = 0.03
	}

	}

	event_building = yes

	ai_allow = {
		# If planet only has slaves/robots, keep it to mining and farming...
		is_slave_tile_or_planet = no
	}
}
