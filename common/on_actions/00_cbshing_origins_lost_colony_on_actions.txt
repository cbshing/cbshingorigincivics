# Triggers event when ship enters a system
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system = {
	events = {
		#cbshing_origins_lost_colony_event.5
	}
}

# A fleet has entered a system
# Scope = Fleet
# From = System
on_entering_system_fleet = {
	events = {
		#cbshing_origins_lost_colony_event.40
		#cbshing_origins_lost_colony_event.41
	}
}

	}
