# Triggers when the game starts
on_game_start_country = {
	events = {
		cbshing_origins_dyson_sphere_start.1 # change species to have habitat preference, spawn habitat, and create new pops there.
	}
}

# A country has increased the level of a tech, use last_increased_tech trigger to check tech and level.
# This = Country
on_tech_increased = {
	events = {
		#cbshing_origins_dyson_sphere_event.1
		#cbshing_origins_dyson_sphere_event.2
	}
}
