namespace = cbshing_origins_eternal_leader_start
# by cbshing

### Hidden Game Start Event
country_event = {
	id = cbshing_origins_eternal_leader_start.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { 
				is_cbshing_eternal_leader_country = yes
			}
			set_country_flag = "origins_eternal_leader_country"
			random_owned_leader = {
				limit = { leader_class = ruler }
				add_trait = leader_trait_origins_ruler_eternal
				add_ruler_trait = leader_trait_origins_ruler_eternal
				set_age = 403
				set_skill = 5
			}	
			if = {
				limit = { 
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
					}
				}
				add_modifier = { modifier = cbshing_origins_eternal_leader_religion_suppressed }
				set_country_flag = "origins_eternal_leader_country_suppressor"
			}
			if = {
				limit = { 
					OR = {
						has_ethic = ethic_spiritualist
						has_ethic = ethic_fanatic_spiritualist
					}
				}
				add_modifier = { modifier = cbshing_origins_eternal_leader_religion_encouraged }
				set_country_flag = "origins_eternal_leader_country_encourager"
			}
		}
	}
}
