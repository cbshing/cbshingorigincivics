### Change homeworld to Non-ideal Start
country_event = {
	id = cbshing_origins_non_ideal_start.1
	hide_window = yes
	is_triggered_only = yes

	trigger = { 
		is_cbshing_different_climate_start_country = yes 
	}

	immediate = {
		random_owned_planet = {
			limit = { is_capital = yes }
			origins_change_planet_type_to_non_ideal_preference = yes
			origins_add_vanilla_tile_blockers = yes
			origins_remove_food_resources = yes
		}
		set_country_flag = "origins_restore_homeworld_needed" 
		add_minerals = 250
	}
}
